    Jetpack Compose 渲染原理深度解析

    一、引言

    Jetpack Compose 是 Google 推出的现代声明式 UI 框架，它彻底改变了 Android UI 的开发方式。与传统的命令式 UI（View 系统）不同，开发者只需描述 UI “应该是什么样子”，而不需要手动管理 UI 的创建、更新和销毁。

    但你是否好奇过：

    • 你写的 @Composable 函数是如何变成屏幕上的像素的？
    • mutableStateOf 的值变化后，Compose 是如何精确地只更新受影响的 UI 的？
    • Slot Table、Applier、Recomposer 这些核心组件各自扮演什么角色？

    本文将基于 AndroidX 源码，深入剖析 Compose 的渲染原理，带你走完从 Composable 函数到屏幕像素的完整旅程。

    三、Composition 阶段：构建 UI 树

    Composition 是 Compose 渲染管线的第一个阶段。在这个阶段，Compose 会执行你的 @Composable 函数，构建出一棵描述 UI 结构的节点树。

public class Recomposer(effectCoroutineContext: CoroutineContext) : CompositionContext() {
    // 状态机：ShutDown → ShuttingDown → Inactive → Idle → PendingWork
    public enum class State {
        ShutDown,        // 已关闭
        ShuttingDown,    // 正在关闭
        Inactive,        // 未激活，等待 runRecomposeAndApplyChanges()
        InactivePendingWork, // 未激活但有待处理的工作
        Idle,            // 空闲，等待新工作

    // Recomposer.kt - 第 565 行
public suspend fun runRecomposeAndApplyChanges(): Unit =
    recompositionRunner { parentFrameClock ->
        val toRecompose = mutableListOf<ControlledComposition>()
        val toApply = mutableListOf<ControlledComposition>()

        while (shouldKeepRecomposing) {
            // 1. 等待有工作可做
            awaitWorkAvailable()

            // 2. 记录 Composer 修改，如果没有待重组的 Composition 则跳过
            if (!recordComposerModifications()) continue

            // 3. 对齐下一帧，合并变更
            parentFrameClock.withFrameNanos { frameTime ->
                // 4. 广播帧时间给动画等订阅者
                broadcastFrameClock.sendFrame(frameTime)

                // 5. 排空快照变更，收集待重组的 Composition
                recordComposerModifications()
                synchronized(stateLock) {
                    compositionInvalidations.forEach { toRecompose += it }
                    compositionInvalidations.clear()
                }

                // 6. 执行重组
                toRecompose.fastForEach { composition ->
                    performRecompose(composition, modifiedValues)?.let {
                        toApply += it
                    }
                }

                // 7. 应用变更到真实 UI 树
                toApply.fastForEach { it.applyChanges() }

                // 8. 应用延迟变更（如可移动内容）
                // 9. 完成变更通知
            }
        }
    }

    关键设计：Recomposer 通过 parentFrameClock.withFrameNanos 与显示帧对齐，确保多次状态变更在同一帧内合并为一次重组，避免不必要的中间状态渲染。

└───┴───┴───┴───┴───┴───┴───┴───┘

    这种策略使得在当前位置的插入和删除操作都是 O(1) 的（摊销），非常适合 Compose 这种线性遍历组合树的场景。

    // AbstractApplier.kt - 第 198 行
public abstract class AbstractApplier<T>(public val root: T) : Applier<T> {
    private val stack = Stack<T>()
    override var current: T = root

    override fun down(node: T) {
        stack.push(current)  // 压栈
        current = node       // 切换到子节点
    }

    override fun up() {
        current = stack.pop()  // 弹栈，回到父节点
    }
}

    // Composables.kt - 第 290 行
@Composable
public inline fun <T : Any, reified E : Applier<*>> ComposeNode(
    noinline factory: () -> T,
    update: @DisallowComposableCalls Updater<T>.() -> Unit,
) {
    if (currentComposer.applier !is E) invalidApplier()

    // 1. 在 Slot Table 中开始一个 Node Group
    currentComposer.startNode()

    if (currentComposer.inserting) {
        // 2a. 首次组合：记录节点创建操作
        currentComposer.createNode(factory)
    } else {
        // 2b. 重组：导航 Applier 到已存在的节点
        currentComposer.useNode()
    }

    // 3. 应用属性更新（每次重组都会执行）
    Updater<T>(currentComposer).update()

    // 4. 结束 Node Group
    currentComposer.endNode()
}

fun draw(canvas: Canvas, graphicsLayer: GraphicsLayer?) {
    val layer = layer
    if (layer != null) {
        // 有缓存层：直接绘制缓存的图形层
        layer.drawLayer(canvas, graphicsLayer)
    } else {
        // 无缓存层：平移 Canvas 到节点位置
        val x = position.x.toFloat()
        val y = position.y.toFloat()
        canvas.translate(x, y)

        // 绘制包含的 DrawModifier
        drawContainedDrawModifiers(canvas, graphicsLayer)

        canvas.translate(-x, -y)
    }
}

// 第 482 行
private fun drawContainedDrawModifiers(canvas: Canvas, graphicsLayer: GraphicsLayer?) {
    val head = head(Nodes.Draw)
    if (head == null) {
        // 没有 DrawModifier：直接绘制子内容
        performDraw(canvas, graphicsLayer)
    } else {
        // 有 DrawModifier：通过 DrawScope 分发
        val drawScope = layoutNode.mDrawScope
        drawScope.draw(canvas, size.toSize(), this, head, graphicsLayer)
    }
}

// 第 492 行
open fun performDraw(canvas: Canvas, graphicsLayer: GraphicsLayer?) {
    // 递归绘制内层 Coordinator
    wrapped?.draw(canvas, graphicsLayer)
}

// 底层实现（简化）
class MutableState<T>(value: T) : State<T> {
    override var value: T
        get() {
            // 读取时：记录当前 RecomposeScope 的观察关系
            Snapshot.current.recordReadOf(this)
            return field
        }
        set(value) {
            // 写入时：触发快照通知
            Snapshot.notifyWrite(this)
            field = value
        }
}

    七、总结

    7.1 关键设计思想

    设计思想体现
    声明式范式描述 UI 应该是什么，而不是如何构建
    单向数据流State → Composition → Layout → Drawing
    精确失效RecomposeScope 粒度的智能重组
    Gap BufferSlot Table 的高效存储策略
    帧对齐合并同一帧内的多次状态变更
    分层架构Runtime（组合）与 UI（布局/绘制）解耦

    7.2 性能优化策略

    1. 跳过未失效的 Composable：通过 RecomposeScope 精确追踪依赖
    2. Gap Buffer：O(1) 的插入/删除（摊销）
    3. 帧合并：多次状态变更在一次帧回调中处理
    4. GraphicsLayer 缓存：避免重复绘制不变的内容
    5. TopDown/BottomUp 选择：根据树特性选择最优插入策略
