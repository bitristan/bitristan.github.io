    Android ART 虚拟机 ClassLinker 深度解析

    概述

    ClassLinker 是 Android ART 运行时的核心组件，负责 Java 类的加载、链接和初始化。它是连接 DEX 文件与运行时内存结构的桥梁，实现了 Java 虚拟机规范中定义的类加载机制。

    本文基于 Android 15 (API 35) 的 ART 源码，深入分析 ClassLinker 的架构设计与实现细节。

    一、ClassLinker 的核心职责

    ClassLinker 的主要职责包括：

    职责描述
    类查找 (FindClass)根据类描述符查找并加载类
    类定义 (DefineClass)从 DEX 文件定义新的类
    类加载 (LoadClass)加载类的字段和方法
    类链接 (LinkClass)解析类的继承关系、实现接口、布局字段
    类初始化 (InitializeClass)执行静态初始化器和父类初始化
    类验证 (VerifyClass)字节码验证确保类型安全
    资源解析解析字符串、类型、方法、字段等

    bool ClassLinker::InitFromBootImage(std::string* error_msg) {
  Runtime* const runtime = Runtime::Current();
  Thread* const self = Thread::Current();
  gc::Heap* const heap = runtime->GetHeap();

  // 获取 Boot Image Spaces
  std::vector<gc::space::ImageSpace*> spaces = heap->GetBootImageSpaces();
  const ImageHeader& image_header = spaces[0]->GetImageHeader();

  // 设置指针大小
  image_pointer_size_ = image_header.GetPointerSize();

  // 从 Image 恢复运行时方法
  runtime->SetResolutionMethod(
      image_header.GetImageMethod(ImageHeader::kResolutionMethod));
  runtime->SetImtConflictMethod(
      image_header.GetImageMethod(ImageHeader::kImtConflictMethod));
  runtime->SetImtUnimplementedMethod(
      image_header.GetImageMethod(ImageHeader::kImtUnimplementedMethod));

  // 恢复 Callee Save 方法
  runtime->SetCalleeSaveMethod(
      image_header.GetImageMethod(ImageHeader::kSaveAllCalleeSavesMethod),
      CalleeSaveType::kSaveAllCalleeSaves);
  // ... 更多 callee save 方法

  // 获取 Class Roots
  class_roots_ = GcRoot<mirror::ObjectArray<mirror::Class>>(
      image_header.GetImageRoot(ImageHeader::kClassRoots));

  // 设置各种 Trampoline
  jni_dlsym_lookup_trampoline_ =
      image_header.GetImageMethod(ImageHeader::kJniDlsymLookup);
  quick_resolution_trampoline_ =
      image_header.GetImageMethod(ImageHeader::kQuickResolutionTrampoline);
  // ...

  return true;
}

    bool ClassLinker::InitWithoutImage(
    std::vector<std::unique_ptr<const DexFile>> boot_class_path,
    std::string* error_msg) {
  Thread* const self = Thread::Current();
  gc::Heap* const heap = runtime->GetHeap();

  // 1. 首先创建 java.lang.Class (它是所有类的类)
  auto class_class_size = mirror::Class::ClassClassSize(image_pointer_size_);
  Handle<mirror::Class> java_lang_Class(hs.NewHandle(
      heap->AllocNonMovableObject(self, nullptr, class_class_size, VoidFunctor())));
  java_lang_Class->SetClass(java_lang_Class.Get());  // Class 的 class 是它自己

  // 2. 创建 Class[] 数组类
  Handle<mirror::Class> class_array_class(hs.NewHandle(
      AllocClass(self, java_lang_Class.Get(),
                 mirror::ObjectArray<mirror::Class>::ClassSize(image_pointer_size_))));
  class_array_class->SetComponentType(java_lang_Class.Get());

  // 3. 创建 java.lang.Object (所有类的超类)
  Handle<mirror::Class> java_lang_Object(hs.NewHandle(
      AllocClass(self, java_lang_Class.Get(),
                 mirror::Object::ClassSize(image_pointer_size_))));
  java_lang_Class->SetSuperClass(java_lang_Object.Get());

  // 4. 创建 Class Roots 数组
  Handle<mirror::ObjectArray<mirror::Class>> class_roots(hs.NewHandle(
      mirror::ObjectArray<mirror::Class>::Alloc(
          self, class_array_class.Get(), ClassRoot::kMax)));

  // 5. 设置 Class Roots
  SetClassRoot(ClassRoot::kJavaLangClass, java_lang_Class.Get());
  SetClassRoot(ClassRoot::kJavaLangObject, java_lang_Object.Get());
  // ...

  // 6. 加载 Boot Class Path
  for (auto& dex_file : boot_class_path) {
    boot_class_path_.push_back(dex_file.get());
    boot_dex_files_.push_back(std::move(dex_file));
  }

  // 7. 创建原始类型类
  CreatePrimitiveClass(self, Primitive::kPrimBoolean, ClassRoot::kPrimitiveBoolean);
  CreatePrimitiveClass(self, Primitive::kPrimByte, ClassRoot::kPrimitiveByte);
  // ...

  // 8. 完成初始化
  FinishInit(self);
  return true;
}

    ObjPtr<mirror::Class> ClassLinker::FindClass(
    Thread* self,
    const char* descriptor,
    size_t descriptor_length,
    Handle<mirror::ClassLoader> class_loader) {

  // 1. 处理原始类型
  if (descriptor_length == 1u) {
    return FindPrimitiveClass(descriptor[0]);
  }

  // 2. 计算描述符哈希
  const size_t hash = ComputeModifiedUtf8Hash(descriptor);

  // 3. 在已加载类表中查找
  ObjPtr<mirror::Class> klass = LookupClass(self, descriptor, hash, class_loader.Get());
  if (klass != nullptr) {
    return EnsureResolved(self, descriptor, klass);
  }

  // 4. 类未加载，需要加载
  if (descriptor[0] != '[' && class_loader == nullptr) {
    // Boot ClassLoader 加载非数组类
    ClassPathEntry pair = FindInClassPath(descriptor, hash, boot_class_path_);
    if (pair.second != nullptr) {
      return DefineClass(self, descriptor, descriptor_length, hash,
                         ScopedNullHandle<mirror::ClassLoader>(),
                         *pair.first, *pair.second);
    }
  } else if (descriptor[0] == '[') {
    // 数组类
    return CreateArrayClass(self, descriptor, descriptor_length, hash, class_loader);
  } else {
    // 应用 ClassLoader
    bool success = FindClassInBaseDexClassLoader(self, descriptor,
                                                  descriptor_length, hash,
                                                  class_loader, &result_ptr);
    if (success && result_ptr != nullptr) {
      return result_ptr;
    }
  }

  return nullptr;  // 类未找到
}

    ObjPtr<mirror::Class> ClassLinker::DefineClass(
    Thread* self,
    const char* descriptor,
    size_t descriptor_length,
    size_t hash,
    Handle<mirror::ClassLoader> class_loader,
    const DexFile& dex_file,
    const dex::ClassDef& dex_class_def) {

  StackHandleScope<3> hs(self);
  Handle<mirror::Class> klass(hs.NewHandle<mirror::Class>(nullptr));

  // 1. 处理特殊核心类 (初始化阶段)
  if (UNLIKELY(!init_done_)) {
    if (descriptor == "Ljava/lang/Object;") {
      klass.Assign(GetClassRoot<mirror::Object>(this));
    } else if (descriptor == "Ljava/lang/Class;") {
      klass.Assign(GetClassRoot<mirror::Class>(this));
    }
    // ...
  }

  // 2. 分配 Class 对象
  if (klass == nullptr) {
    klass.Assign(AllocClass(self, SizeOfClassWithoutEmbeddedTables(dex_file, dex_class_def)));
  }

  // 3. 设置类基本信息
  SetupClass(dex_file, dex_class_def, klass, class_loader);

  // 4. 注册到 ClassTable (可能在其他线程被访问)
  ObjPtr<mirror::Class> existing = InsertClass(descriptor, klass.Get(), hash);
  if (existing != nullptr) {
    // 已有同名类，返回已有的
    return EnsureResolved(self, descriptor, existing);
  }

  // 5. 加载类内容 (字段、方法)
  LoadClass(self, dex_file, dex_class_def, klass);

  // 6. 链接类
  MutableHandle<mirror::Class> h_new_class(hs.NewHandle<mirror::Class>(nullptr));
  if (!LinkClass(self, descriptor, klass, interfaces, &h_new_class)) {
    return nullptr;
  }

  // 7. 返回链接后的类
  return h_new_class.Get();
}

    void ClassLinker::LoadClass(
    Thread* self,
    const DexFile& dex_file,
    const dex::ClassDef& dex_class_def,
    Handle<mirror::Class> klass) {

  ClassAccessor accessor(dex_file, dex_class_def);

  // 禁止线程挂起，确保字段和方法数组完整
  ScopedAssertNoThreadSuspension nts(__FUNCTION__);

  // 获取分配器
  LinearAlloc* const allocator = GetAllocatorForClassLoader(klass->GetClassLoader());

  // 1. 分配静态字段数组
  LengthPrefixedArray<ArtField>* sfields = AllocArtFieldArray(
      self, allocator, accessor.NumStaticFields());

  // 2. 分配实例字段数组
  LengthPrefixedArray<ArtField>* ifields = AllocArtFieldArray(
      self, allocator, accessor.NumInstanceFields());

  // 3. 分配方法数组
  LengthPrefixedArray<ArtMethod>* methods = AllocArtMethodArray(
      self, allocator, accessor.NumMethods());

  // 4. 加载字段和方法
  accessor.VisitFieldsAndMethods(
      [&](const ClassAccessor::Field& field) {
        // 加载静态字段
        LoadField(field, klass, &sfields->At(num_sfields++));
      },
      [&](const ClassAccessor::Field& field) {
        // 加载实例字段
        LoadField(field, klass, &ifields->At(num_ifields++));
      },
      [&](const ClassAccessor::Method& method) {
        // 加载方法
        LoadMethod(dex_file, method, klass, &mai_direct, &methods->At(method_idx));
        LinkCode(&methods->At(method_idx), class_def_method_index, &occi);
      });

  // 5. 设置类的字段和方法指针
  klass->SetSFieldsPtr(sfields);
  klass->SetIFieldsPtr(ifields);
  klass->SetMethodsPtr(methods, accessor.NumDirectMethods(), accessor.NumVirtualMethods());
}

    bool ClassLinker::LinkClass(
    Thread* self,
    const char* descriptor,
    Handle<mirror::Class> klass,
    Handle<mirror::ObjectArray<mirror::Class>> interfaces,
    MutableHandle<mirror::Class>* h_new_class_out) {

  CHECK_EQ(ClassStatus::kLoaded, klass->GetStatus());

  // 1. 链接父类
  if (!LinkSuperClass(klass)) {
    return false;
  }

  // 2. 链接方法 (创建 vtable 和 itable)
  ArtMethod* imt_data[ImTable::kSize];
  bool new_conflict = false;
  if (!LinkMethods(self, klass, interfaces, &new_conflict, imt_data)) {
    return false;
  }

  // 3. 链接实例字段 (计算字段偏移)
  if (!LinkInstanceFields(self, klass)) {
    return false;
  }

  // 4. 链接静态字段 (计算类大小)
  size_t class_size;
  if (!LinkStaticFields(self, klass, &class_size)) {
    return false;
  }

  // 5. 创建 IMT (Interface Method Table)
  ImTable* imt = nullptr;
  if (klass->ShouldHaveImt()) {
    // 尝试复用父类的 IMT
    if (!new_conflict) {
      ImTable* super_imt = klass->FindSuperImt(image_pointer_size_);
      if (super_imt != nullptr && ImtEquals(super_imt, imt_data)) {
        imt = super_imt;
      }
    }
    if (imt == nullptr) {
      imt = CreateImt(self, klass, imt_data);
    }
  }

  // 6. 如果类大小变化，需要重新分配
  if (class_size != klass->GetClassSize()) {
    klass.Assign(MirrorClass(klass.Get(), class_size, imt));
  } else {
    klass->SetImt(imt, image_pointer_size_);
  }

  // 7. 更新类状态为已解析
  klass->SetStatus(ClassStatus::kResolved, self);

  h_new_class_out->Assign(klass.Get());
  return true;
}

    bool ClassLinker::InitializeClass(
    Thread* self,
    Handle<mirror::Class> klass,
    bool can_init_statics,
    bool can_init_parents) {

  // 1. 检查是否已初始化
  if (klass->IsInitialized()) {
    return true;
  }

  // 2. 获取类锁
  ObjectLock<mirror::Class> lock(self, klass);

  // 3. 再次检查 (double-check locking)
  if (klass->IsInitialized()) {
    return true;
  }

  // 4. 检查是否为错误状态
  if (klass->IsErroneous()) {
    ThrowEarlierClassFailure(klass.Get(), true);
    return false;
  }

  // 5. 验证类
  if (!klass->IsVerified()) {
    VerifyClass(self, nullptr, klass);
    if (!klass->IsVerified()) {
      return false;
    }
  }

  // 6. 初始化父类
  if (can_init_parents && klass->GetSuperClass() != nullptr) {
    if (!EnsureInitialized(self, klass->GetSuperClass(), true, true)) {
      return false;
    }
  }

  // 7. 设置状态为初始化中
  klass->SetStatus(ClassStatus::kInitializing, self);

  // 8. 初始化静态字段
  if (can_init_statics) {
    InitializeStaticFields(klass);
  }

  // 9. 执行 <clinit> 方法
  ArtMethod* clinit = klass->GetClassInitializer();
  if (clinit != nullptr) {
    // 执行静态初始化器
    clinit->Invoke(self, ...);
  }

  // 10. 设置状态为已初始化
  klass->SetStatus(ClassStatus::kInitialized, self);

  return true;
}

    ObjPtr<mirror::Class> ClassLinker::ResolveType(
    dex::TypeIndex type_idx,
    Handle<mirror::DexCache> dex_cache,
    Handle<mirror::ClassLoader> class_loader) {

  // 1. 检查 DexCache 缓存
  ObjPtr<mirror::Class> resolved = dex_cache->GetResolvedType(type_idx);
  if (resolved != nullptr) {
    return resolved;
  }

  // 2. 未缓存，执行解析
  return DoResolveType(type_idx, dex_cache, class_loader);
}

ObjPtr<mirror::Class> ClassLinker::DoResolveType(
    dex::TypeIndex type_idx,
    Handle<mirror::DexCache> dex_cache,
    Handle<mirror::ClassLoader> class_loader) {

  const DexFile& dex_file = *dex_cache->GetDexFile();
  const char* descriptor = dex_file.GetTypeDescriptor(type_idx);

  // 查找类
  ObjPtr<mirror::Class> klass = FindClass(self, descriptor, class_loader);

  // 缓存结果
  if (klass != nullptr) {
    dex_cache->SetResolvedType(type_idx, klass);
  }

  return klass;
}

    void ClassLinker::CleanupClassLoaders() {
  Thread* const self = Thread::Current();
  MutexLock mu(self, Locks::classlinker_classes_lock_);

  for (auto it = class_loaders_.begin(); it != class_loaders_.end(); ) {
    ClassLoaderData& data = *it;

    // 检查弱引用是否被清除
    if (data.weak_root == nullptr) {
      // 准备删除
      PrepareToDeleteClassLoader(self, data, /*cleanup_cha=*/true);

      // 删除 ClassTable 和 LinearAlloc
      delete data.class_table;
      delete data.allocator;

      it = class_loaders_.erase(it);
    } else {
      ++it;
    }
  }
}

    void ClassLinker::RemoveDexFromCaches(const DexFile& dex_file) {
  Thread* const self = Thread::Current();
  MutexLock mu(self, Locks::dex_lock_);

  auto it = dex_caches_.find(&dex_file);
  if (it != dex_caches_.end()) {

  }
}

    ObjPtr<mirror::Class> ClassLinker::CreateProxyClass(
    ScopedObjectAccessAlreadyRunnable& soa,
    jstring name,
    jobjectArray interfaces,
    jobject loader,
    jobjectArray methods,
    jobjectArray throws) {

  Thread* self = soa.Self();

  // 1. 分配 Proxy 类
  Handle<mirror::Class> klass = hs.NewHandle(
      AllocClass(self, mirror::Proxy::ProxyClassSize(image_pointer_size_)));

  // 2. 设置类属性
  klass->SetAccessFlags(kAccPublic | kAccFinal | kAccAbstract);
  klass->SetClassLoader(loader);
  klass->SetDexCache(dex_cache);

  // 3. 设置父类为 java.lang.reflect.Proxy
  klass->SetSuperClass(GetClassRoot<mirror::Proxy>(this));

  // 4. 创建方法
  for (size_t i = 0; i < num_methods; ++i) {
    CreateProxyMethod(klass, prototype, &methods->At(i));
  }

  // 5. 链接类
  LinkClass(self, descriptor, klass, interfaces, &h_new_class);

  return h_new_class.Get();
}

    十四、总结

    ClassLinker 是 ART 运行时的核心组件，实现了完整的 Java 类加载机制：

    1. 类查找与加载：支持 Boot ClassLoader 和应用 ClassLoader
    2. 类链接：解析继承关系、布局字段、创建方法表
    3. 类初始化：遵循 JLS 规范的初始化流程
    4. 性能优化：IMT、CHA、缓存机制
    5. 内存管理：支持类卸载、LinearAlloc

    理解 ClassLinker 对于深入理解 ART 虚拟机和 Android 应用启动性能至关重要。
