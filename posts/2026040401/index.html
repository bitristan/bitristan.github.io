                displayData.vsyncTraceToggle);
    
    return displayIdOpt;
}

    // VSyncReactor.cpp:154-189
bool VSyncReactor::periodConfirmed(nsecs_t vsync_timestamp, 
                                    std::optional<nsecs_t> HwcVsyncPeriod) {
    // еҝ…йЎ»жңүдёҠж¬Ўи®°еҪ•жҲ– HWC жҸҗдҫӣзҡ„е‘Ёжңҹ
    if (!mLastHwVsync && !HwcVsyncPeriod) {
        return false;  // ж ·жң¬дёҚи¶і
    }
    
    // жЈҖжҹҘе‘ЁжңҹжҳҜеҗҰеңЁе…Ғи®ёиҜҜе·®иҢғеӣҙеҶ…пјҲ10%пјү
    auto const allowance = period * 0.1;
    if (HwcVsyncPeriod) {
        return std::abs(*HwcVsyncPeriod - period) < allowance;
    }
    
    // жҲ–иҖ…йҖҡиҝҮдёӨж¬Ўж—¶й—ҙжҲіе·®еҖји®Ўз®—
    auto const distance = vsync_timestamp - *mLastHwVsync;
    return std::abs(distance - period) < allowance;
}

void EventThread::threadMain(std::unique_lock<std::mutex>& lock) {
    while (mState != State::Quit) {
        std::vector<sp<EventThreadConnection>> connections;
        
        // ж”¶йӣҶйңҖиҰҒ VSYNC зҡ„иҝһжҺҘ

            if (connection->vsyncRequest != VSyncRequest::None) {
                connections.push_back(connection);
            }
        }
        
        // зӯүеҫ… VSYNC дҝЎеҸ·

        // еҲҶеҸ‘ VSYNC дәӢд»¶
        for (const auto& connection : connections) {
            connection->postEvent(vsyncEvent);
        }
    }
}
